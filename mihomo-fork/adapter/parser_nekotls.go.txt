package adapter

	"fmt"

	"github.com/metacubex/mihomo/adapter/outbound"
	"github.com/metacubex/mihomo/common/structure"
	C "github.com/metacubex/mihomo/constant"
)

// ParseProxy: add this case inside the existing switch statement.
func ParseProxy(mapping map[string]any) (C.Proxy, error) {
	decoder := structure.NewDecoder(structure.Option{TagName: "proxy", WeaklyTypedInput: true})
	proxyType, existType := mapping["type"].(string)
	if !existType {
		return nil, fmt.Errorf("missing type")
	}

	switch proxyType {
	// ... existing cases (vless, trojan, anytls, ...) ...

	case "nekotls":
		nekoOption := &outbound.NekoTLSOption{}
		if err := decoder.Decode(mapping, nekoOption); err != nil {
			return nil, err
		}
		proxy, err := outbound.NewNekoTLS(*nekoOption)
		if err != nil {
			return nil, err
		}
		return NewProxy(proxy), nil

	default:
		return nil, fmt.Errorf("unsupport proxy type: %s", proxyType)
	}
}
